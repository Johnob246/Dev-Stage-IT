Class topLevelWindowFindRecipe:
    Method __init__(parent):
        Call super().__init__(parent)
        Set title to "Find Recipe"
        Set geometry to 600x500
        Set minsize to 600x500
        Set maxsize to 600x500
        Initialize topLevelViewRecipe to None
        
    Method listUpdate():
        Set x to 0
        For each directory in list directories in "Recipe Storage":
            Call listBoxFindRecipe.delete(x)
            Increment x by 1
        Set x to 0
        For each directory2 in list directories in "Recipe Storage":
            If entryFindRecipeSearchBar.get() in directory2.lower():
                Call listBoxFindRecipe.insert(x, directory2)
            Increment x by 1

    Method listGet(num):
        Call entryFindRecipeSearchBar.delete(0, END)
        Call entryFindRecipeSearchBar.insert(index=0, string=f"{num}")

    Method addRecipeWindowAttributes2():
        Create CTkLabel with bg_color="purple", text=""
        Pack label with expand=True, fill="both"
        Create lblFindRecipeTitle as CTkLabel with corner_radius=32, bg_color="purple", fg_color="black", text="Find Recipe", font=("Arial", 24)
        Create btnViewRecipe2 as CTkButton with corner_radius=32, border_width=2, bg_color="purple", fg_color="black", border_color="white", text="View Recipe", font=("Arial", 20), command=lambda: windowOpenViewRecipe()
        Create btnExit2 as CTkButton with corner_radius=32, border_width=2, bg_color="purple", fg_color="black", border_color="white", text="Exit", font=("Arial", 20), command=self.destroy
        Create lblFindRecipeSearchBar as CTkLabel with corner_radius=32, bg_color="purple", fg_color="black", text="Search Recipe", font=("Arial", 15)
        Create entryFindRecipeSearchBar as CTkEntry with corner_radius=32, bg_color="purple", fg_color="black", width=300
        Create listBoxFindRecipe as CTkListbox with bg_color="purple", fg_color="black", height=200, width=300, command=self.listGet

    Method placeRecipeWindowAttributes2():
        Place lblFindRecipeTitle at (220, 13)
        Place btnViewRecipe2 at (440, 460)
        Place btnExit2 at (13, 460)
        Place lblFindRecipeSearchBar at (20, 50)
        Place entryFindRecipeSearchBar at (150, 50)
        Place listBoxFindRecipe at (120, 100)
        Initialize indexValue to 0
        For each directory in list directories in "Recipe Storage":
            Call listBoxFindRecipe.insert(indexValue, directory)
            Increment indexValue by 1
        Bind "<KeyRelease>" event to listUpdate

    Method windowOpenViewRecipe():
        If topLevelViewRecipe is None or topLevelViewRecipe does not exist:
            Set topLevelViewRecipe to new topLevelWindowViewRecipe(self)
            Call topLevelViewRecipe.viewRecipeWindowAttributes()
            Call topLevelViewRecipe.placeViewRecipeWindowAttributes()
        Else:
            Focus topLevelViewRecipe
